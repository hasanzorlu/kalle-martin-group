CREATE TABLE Person(CPR VARCHAR(13),type enum('elder','admin'),pictureURL VARCHAR(150),PRIMARY KEY(CPR));

CREATE TABLE Clothes(id VARCHAR(30),type enum('Sweater','Shirt','Pants','Underwear','WashingBag','Dress','Shorts','Towel','BedLinen'),color enum('Colored','White'),washingTemperature enum('Degree30','Degree40','Degree60','Degree95','Handwash'),dryRecipe enum('HangDry','Tumbling'),fabric enum('Cotton','Synthetic','Wool'),weight INT, state INT, binId INT,CPRNo VARCHAR(13),PRIMARY KEY(id),FOREIGN KEY (CPRNo) REFERENCES Person (CPR));

CREATE TABLE Tags(Id VARCHAR(30),YEAR decimal(4),MONTH decimal(2),DAY decimal(2),washCount INT, CPRno VARCHAR(13),PRIMARY KEY (Id),FOREIGN KEY (CPRNo) REFERENCES Person (CPR));

CREATE TABLE Name(name VARCHAR(50),surname VARCHAR(30),middleName VARCHAR(30),CPRno VARCHAR(13),FOREIGN KEY (CPRNo) REFERENCES Person (CPR));

CREATE TABLE Address(street VARCHAR(30), buildingNo VARCHAR(30), roomNo VARCHAR(30), city VARCHAR(30), postCode VARCHAR(30), country VARCHAR(30), CPRno VARCHAR(13),PRIMARY KEY(CPRno), FOREIGN KEY (CPRNo) REFERENCES Person (CPR));
